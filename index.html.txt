  const persian = ['۰','۱','۲','۳','۴','۵','۶','۷','۸','۹'];
  return String(num).replace(/\d/g, d => persian[d]);
}

function applyTheme() {
  document.documentElement.setAttribute('data-theme', state.theme);
  const icon = document.getElementById('theme-icon');
  if (state.theme === 'dark') {
    icon.innerHTML = '<circle cx="12" cy="12" r="5"/><path d="M12 1v2M12 21v2M4.22 4.22l1.42 1.42M18.36 18.36l1.42 1.42M1 12h2M21 12h2M4.22 19.78l1.42-1.42M18.36 5.64l1.42-1.42"/>';
  } else {
    icon.innerHTML = '<path d="M21 12.79A9 9 0 1111.21 3 7 7 0 0021 12.79z"/>';
  }
}
function toggleTheme() {
  state.theme = state.theme === 'light' ? 'dark' : 'light';
  localStorage.setItem('luxe_theme_fa', state.theme);
  applyTheme();
}

function navigate(page) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.getElementById('page-' + page).classList.add('active');
  document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));
  const link = document.querySelector(`.nav-link[data-page="${page}"]`);
  if (link) link.classList.add('active');
  state.currentPage = page;
  window.scrollTo({ top: 0, behavior: 'smooth' });
  if (page === 'cart') renderCart();
  if (page === 'wishlist') renderWishlist();
  if (page === 'checkout') renderCheckout();
  if (page === 'shop') renderShop();
}

function toggleMobileMenu() {
  document.getElementById('mobile-menu').classList.toggle('active');
}

  const inWishlist = state.wishlist.includes(p.id);
  const price = p.flash ? p.flashPrice : p.price;

  return `

      <div class="product-image">
        <img src="${p.image}" alt="${p.name}" loading="lazy" />
        ${p.flash ? `<div class="product-tag flash">${toPersianNum(discount)}٪-</div>` : ''}
        <button class="product-wishlist ${inWishlist ? 'active' : ''}" onclick="event.stopPropagation(); toggleWishlist(${p.id})">

        </button>
        <div class="product-quick-add" onclick="event.stopPropagation(); quickAdd(${p.id})">افزودن سریع +</div>
      </div>
      <div class="product-info">
        <div class="product-brand">${p.brand}</div>
        <div class="product-name">${p.name}</div>
        <div class="product-rating">

        </div>

function renderFeatured() {
  const featured = PRODUCTS.filter(p => p.featured).slice(0, 4);
  document.getElementById('featured-products').innerHTML = featured.map(createProductCard).join('');
}

function renderShop() {
  let filtered = [...PRODUCTS];
  const f = state.filters;
  if (f.category) filtered = filtered.filter(p => p.category === f.category);
  if (f.search) {
    const s = f.search.toLowerCase();
    filtered = filtered.filter(p => p.name.toLowerCase().includes(s) || p.brand.toLowerCase().includes(s) || p.category.toLowerCase().includes(s));
  }
  filtered = filtered.filter(p => p.price >= f.minPrice && p.price <= f.maxPrice);
  if (f.rating > 0) filtered = filtered.filter(p => p.rating >= f.rating);
  if (f.sort === 'price-low') filtered.sort((a, b) => a.price - b.price);
  if (f.sort === 'price-high') filtered.sort((a, b) => b.price - a.price);
  if (f.sort === 'rating') filtered.sort((a, b) => b.rating - a.rating);
  if (f.sort === 'newest') filtered.sort((a, b) => b.id - a.id);
  document.getElementById('shop-products').innerHTML = filtered.length ? filtered.map(createProductCard).join('') : '<p style="grid-column: 1/-1; text-align: center; padding: 80px 24px; color: var(--text-muted);">هیچ محصولی با معیارهای شما مطابقت ندارد.</p>';
  document.getElementById('shop-count').textContent = `نمایش ${toPersianNum(filtered.length)} محصول`;
}

function renderCategoryFilters() {
  const categories = [...new Set(PRODUCTS.map(p => p.category))];
  document.getElementById('category-filters').innerHTML = categories.map(c => `
    <label class="filter-option">
      <input type="checkbox" value="${c}" onchange="toggleCategory('${c}')" ${state.filters.category === c ? 'checked' : ''} />
      ${c}
    </label>
  `).join('');
}

function toggleCategory(cat) {
  state.filters.category = state.filters.category === cat ? null : cat;
  document.querySelectorAll('#category-filters input').forEach(i => i.checked = (i.value === state.filters.category));
  renderShop();
}

function filterCategory(cat) {
  state.filters.category = cat;
  navigate('shop');
  renderCategoryFilters();
  renderShop();
}

function handleSearch(val) {
  state.filters.search = val;
  renderShop();
}

function applyFilters() {
  state.filters.minPrice = parseFloat(document.getElementById('price-min').value) || 0;
  state.filters.maxPrice = parseFloat(document.getElementById('price-max').value) || 2000000;
  const ratingEl = document.querySelector('input[name="rating"]:checked');
  state.filters.rating = ratingEl ? parseFloat(ratingEl.value) : 0;
  state.filters.sort = document.getElementById('shop-sort').value;
  renderShop();
}

function resetFilters() {
  state.filters = { category: null, minPrice: 0, maxPrice: 2000000, rating: 0, search: '', sort: 'featured' };
  document.getElementById('search-input').value = '';
  document.getElementById('price-min').value = 0;
  document.getElementById('price-max').value = 2000000;
  document.querySelector('input[name="rating"][value="0"]').checked = true;
  document.getElementById('shop-sort').value = 'featured';
  renderCategoryFilters();
  renderShop();
}

function viewProduct(id) {
  const p = PRODUCTS.find(x => x.id === id);
  if (!p) return;
  state.currentProduct = p;
  state.quantity = 1;
  state.selectedShade = 0;
  state.selectedSize = 0;
  document.getElementById('pd-main-image').src = p.image;
  document.getElementById('pd-brand').textContent = p.brand;
  document.getElementById('pd-name').textContent = p.name;
  document.getElementById('pd-stars').textContent = '★'.repeat(Math.floor(p.rating));
  document.getElementById('pd-rating-text').textContent = `${toPersianNum(p.rating)} · ${toPersianNum(p.reviews)} نظر`;
  const price = p.flash ? p.flashPrice : p.price;
  document.getElementById('pd-price').textContent = formatPrice(price);
  document.getElementById('pd-original').textContent = p.originalPrice > price ? formatPrice(p.originalPrice) : '';
  const disc = p.originalPrice > price ? '-' + Math.round((1 - price / p.originalPrice) * 100) + '٪' : '';
  document.getElementById('pd-discount').textContent = disc;
  document.getElementById('pd-desc').textContent = p.description;
  document.getElementById('pd-sku').textContent = p.sku;
  document.getElementById('pd-cat').textContent = p.category;
  document.getElementById('pd-qty').textContent = toPersianNum(1);
  document.getElementById('pd-shades').innerHTML = p.shades.length ? p.shades.map((s, i) => `<div class="shade ${i === 0 ? 'active' : ''}" style="background:${s}" onclick="selectShade(${i})"></div>`).join('') : '<span style="color: var(--text-muted); font-size: 0.9rem;">یک سایز</span>';
  document.getElementById('pd-sizes').innerHTML = p.sizes.map((s, i) => `<button class="size-btn ${i === 0 ? 'active' : ''}" onclick="selectSize(${i})">${s}</button>`).join('');
  document.getElementById('pd-thumbnails').innerHTML = [p.image, p.image, p.image, p.image].map((img, i) => `<div class="product-thumbnail ${i === 0 ? 'active' : ''}" onclick="selectThumbnail(this, '${img}')"><img src="${img}" /></div>`).join('');
  document.getElementById('reviews-list').innerHTML = REVIEWS.map(r => `
    <div class="review-card">
      <div class="review-header">
        <div><span class="review-author">${r.author}</span> <span class="stars" style="margin-right: 8px;">${'★'.repeat(r.rating)}</span></div>
        <div class="review-date">${r.date}</div>
      </div>
      <p class="review-text">${r.text}</p>
    </div>
  `).join('');
  const recs = PRODUCTS.filter(x => x.id !== p.id).slice(0, 4);
  document.getElementById('recommended-products').innerHTML = recs.map(createProductCard).join('');
  navigate('product');
}

  state.selectedShade = i;
  document.querySelectorAll('#pd-shades .shade').forEach((el, idx) => el.classList.toggle('active', idx === i));
}
function selectSize(i) {
  state.selectedSize = i;
  document.querySelectorAll('#pd-sizes .size-btn').forEach((el, idx) => el.classList.toggle('active', idx === i));

function changeQty(delta) {
  state.quantity = Math.max(1, Math.min(10, state.quantity + delta));
  document.getElementById('pd-qty').textContent = toPersianNum(state.quantity);
}

function addToCartFromDetail() {
  if (!state.currentProduct) return;
  addToCart(state.currentProduct.id, state.quantity);
}

function toggleWishlistFromDetail() {
  if (!state.currentProduct) return;
  toggleWishlist(state.currentProduct.id);
}

function addToCart(id, qty = 1) {
  const existing = state.cart.find(i => i.id === id);
  if (existing) existing.qty += qty;
  else state.cart.push({ id, qty });
  saveCart();
  updateBadges();
  showToast('به سبد خرید اضافه شد', 'success');
}

function quickAdd(id) { addToCart(id, 1); }

function removeFromCart(id) {
  state.cart = state.cart.filter(i => i.id !== id);
  saveCart();
  renderCart();
  updateBadges();
  showToast('از سبد حذف شد', 'success');
}

function updateCartQty(id, delta) {
  const item = state.cart.find(i => i.id === id);
  if (!item) return;
  item.qty = Math.max(1, item.qty + delta);
  saveCart();
  renderCart();
  updateBadges();
}

function saveCart() { localStorage.setItem('luxe_cart_fa', JSON.stringify(state.cart)); }

function renderCart() {
  const container = document.getElementById('cart-items');
  const empty = document.getElementById('cart-empty');
  if (state.cart.length === 0) {
    container.innerHTML = '';
    empty.style.display = 'block';
    document.getElementById('cart-count-text').textContent = '۰ کالا در سبد شما';
    updateCartTotals(0);
    return;
  }
  empty.style.display = 'none';
  let subtotal = 0;
  container.innerHTML = state.cart.map(item => {
    const p = PRODUCTS.find(x => x.id === item.id);
    if (!p) return '';
    const price = p.flash ? p.flashPrice : p.price;
    subtotal += price * item.qty;
    return `
      <div class="cart-item">
        <div class="cart-item-image"><img src="${p.image}" /></div>
        <div class="cart-item-info">
          <div class="product-brand">${p.brand}</div>
          <h3>${p.name}</h3>
          <div style="font-size: 0.85rem; color: var(--text-muted); margin-top: 4px;">${p.category} · ${p.sizes[state.selectedSize] || p.sizes[0]}</div>
          <div class="quantity-selector" style="margin-top: 12px;">
            <button class="quantity-btn" onclick="updateCartQty(${p.id}, -1)">−</button>
            <div class="quantity-value">${toPersianNum(item.qty)}</div>
            <button class="quantity-btn" onclick="updateCartQty(${p.id}, 1)">+</button>
          </div>
          <button class="cart-item-remove" onclick="removeFromCart(${p.id})">حذف</button>
        </div>
        <div class="cart-item-price">${formatPrice(price * item.qty)}</div>
      </div>`;
  }).join('');
  const totalItems = state.cart.reduce((s, i) => s + i.qty, 0);
  document.getElementById('cart-count-text').textContent = `${toPersianNum(totalItems)} کالا در سبد شما`;
  updateCartTotals(subtotal);
}

function updateCartTotals(subtotal) {
  const shipping = subtotal > 500000 ? 0 : 120000;
  const discount = state.coupon === 'LUXE20' ? subtotal * 0.2 : (state.coupon === 'SAVE10' ? 100000 : 0);
  const afterDiscount = Math.max(0, subtotal - discount);
  const tax = afterDiscount * 0.09;
  const total = afterDiscount + shipping + tax;
  document.getElementById('cart-subtotal').textContent = formatPrice(subtotal);
  document.getElementById('cart-shipping').textContent = shipping === 0 ? 'رایگان' : formatPrice(shipping);
  document.getElementById('cart-tax').textContent = formatPrice(tax);
  document.getElementById('cart-total').textContent = formatPrice(total);
  const discRow = document.getElementById('discount-row');
  if (discount > 0) {
    discRow.style.display = 'flex';
    document.getElementById('cart-discount').textContent = '-' + formatPrice(discount);
  } else discRow.style.display = 'none';
}

function applyCoupon() {
  const code = document.getElementById('coupon-input').value.trim().toUpperCase();
  const msg = document.getElementById('coupon-message');
  if (code === 'LUXE20') {
    state.coupon = code;
    msg.textContent = '✓ تخفیف ۲۰٪ اعمال شد!';
    msg.style.color = 'var(--success)';
    showToast('کد تخفیف اعمال شد: ۲۰٪ تخفیف', 'success');
  } else if (code === 'SAVE10') {
    state.coupon = code;
    msg.textContent = '✓ تخفیف ۱۰۰,۰۰۰ تومانی اعمال شد!';
    msg.style.color = 'var(--success)';
    showToast('کد تخفیف اعمال شد: ۱۰۰ هزار تومان تخفیف', 'success');
  } else {
    state.coupon = null;
    msg.textContent = '✗ کد تخفیف نامعتبر است';
    msg.style.color = 'var(--danger)';
    showToast('کد تخفیف نامعتبر', 'error');
  }
  renderCart();
}

function toggleWishlist(id) {
  const idx = state.wishlist.indexOf(id);
  if (idx > -1) {
    state.wishlist.splice(idx, 1);
    showToast('از علاقه‌مندی‌ها حذف شد', 'success');
  } else {
    state.wishlist.push(id);
    showToast('به علاقه‌مندی‌ها اضافه شد', 'success');
  }
  localStorage.setItem('luxe_wishlist_fa', JSON.stringify(state.wishlist));
  updateBadges();
  renderFeatured();
  renderShop();
  if (state.currentPage === 'wishlist') renderWishlist();
}

function renderWishlist() {
  const container = document.getElementById('wishlist-products');
  const empty = document.getElementById('wishlist-empty');
  if (state.wishlist.length === 0) {
    container.innerHTML = '';
    empty.style.display = 'block';
    return;
  }
  empty.style.display = 'none';
  const items = state.wishlist.map(id => PRODUCTS.find(p => p.id === id)).filter(Boolean);
  container.innerHTML = items.map(createProductCard).join('');
}

function updateBadges() {
  document.getElementById('cart-badge').textContent = toPersianNum(state.cart.reduce((s, i) => s + i.qty, 0));
  document.getElementById('wishlist-badge').textContent = toPersianNum(state.wishlist.length);
}

function renderCheckout() {
  let subtotal = 0;
  document.getElementById('checkout-items').innerHTML = state.cart.map(item => {
    const p = PRODUCTS.find(x => x.id === item.id);
    if (!p) return '';
    const price = p.flash ? p.flashPrice : p.price;
    subtotal += price * item.qty;
    return `<div style="display: flex; gap: 12px; padding: 12px 0; border-bottom: 1px solid var(--border);">
      <img src="${p.image}" style="width: 60px; height: 60px; object-fit: cover; border-radius: 4px;" />
      <div style="flex: 1;">
        <div style="font-weight: 600; font-size: 0.9rem;">${p.name}</div>
        <div style="font-size: 0.8rem; color: var(--text-muted);">تعداد: ${toPersianNum(item.qty)}</div>
      </div>
      <div style="font-weight: 600;">${formatPrice(price * item.qty)}</div>
    </div>`;
  }).join('');
  const shipping = 120000;
  const tax = subtotal * 0.09;
  const total = subtotal + shipping + tax;
  document.getElementById('checkout-subtotal').textContent = formatPrice(subtotal);
  document.getElementById('checkout-tax').textContent = formatPrice(tax);
  document.getElementById('checkout-total').textContent = formatPrice(total);
}

function placeOrder() {
  if (state.cart.length === 0) { showToast('سبد خرید شما خالی است', 'error'); return; }
  showToast('سفارش با موفقیت ثبت شد! 🎉', 'success');
  state.cart = [];
  saveCart();
  updateBadges();
  setTimeout(() => navigate('account'), 1500);
}

function openAuthModal() { document.getElementById('auth-modal').classList.add('active'); }
function closeAuthModal() { document.getElementById('auth-modal').classList.remove('active'); }
function switchAuthTab(el, tab) {
  document.querySelectorAll('.auth-tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
  document.getElementById('auth-register-name').style.display = tab === 'register' ? 'block' : 'none';
  document.getElementById('auth-submit').textContent = tab === 'register' ? 'ایجاد حساب' : 'ورود';
}
function handleAuth(e) {
  e.preventDefault();
  closeAuthModal();
  showToast('به لوکس خوش آمدید! ✨', 'success');
}

function showToast(msg, type = 'success') {
  const container = document.getElementById('toast-container');
  const toast = document.createElement('div');
  toast.className = `toast ${type}`;
  toast.innerHTML = `<span>${type === 'success' ? '✓' : '✗'}</span><span>${msg}</span>`;
  container.appendChild(toast);
  setTimeout(() => { toast.style.opacity = '0'; toast.style.transform = 'translateX(-100px)'; toast.style.transition = 'all 0.3s'; setTimeout(() => toast.remove(), 300); }, 3000);
}

function startCountdown() {
  const end = Date.now() + 2 * 24 * 60 * 60 * 1000 + 14 * 60 * 60 * 1000 + 32 * 60 * 1000 + 18 * 1000;
  setInterval(() => {
    const diff = Math.max(0, end - Date.now());
    const d = Math.floor(diff / (1000 * 60 * 60 * 24));
    const h = Math.floor((diff / (1000 * 60 * 60)) % 24);
    const m = Math.floor((diff / (1000 * 60)) % 60);
    const s = Math.floor((diff / 1000) % 60);
    document.getElementById('cd-days').textContent = toPersianNum(String(d).padStart(2, '0'));
    document.getElementById('cd-hours').textContent = toPersianNum(String(h).padStart(2, '0'));
    document.getElementById('cd-mins').textContent = toPersianNum(String(m).padStart(2, '0'));
    document.getElementById('cd-secs').textContent = toPersianNum(String(s).padStart(2, '0'));
  }, 1000);
}
